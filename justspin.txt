#include <AccelStepper.h>

const int dirPin = 12;
const int stepPin = 14;


#define motorInterfaceType 1


AccelStepper myStepper(motorInterfaceType, stepPin, dirPin);

void setup() {
    myStepper.setMaxSpeed(700);
   
    myStepper.setAcceleration(300);
    
    myStepper.moveTo(10000000); 
}

void loop() {
    // This must be called continuously to make the motor move
    myStepper.run();
}
